'use strict';

/** @function
 * @name bazinga
 * @param {number} foo Un número formidable.
 * @returns {boolean} Si foo es 1, devuelve true.
 * @description Esta función comprueba si foo es 1 o no
 */

function bazinga(foo) {
    if (foo === 1) {
        return true;
    }
    return false;
}
